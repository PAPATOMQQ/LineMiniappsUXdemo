// Function to display area-specific page
function showArea(area) {
    const areaData = mockData[area];
    const main = document.querySelector('main');
    main.innerHTML = `
        <h2>${areaData.area}</h2>
        <p id="last-updated">Last Updated: ${new Date().toLocaleTimeString()}</p>
        <ul>
            ${areaData.locations.map(loc => `
                <li>
                    <span class="dot ${loc.color}"></span>
                    ${loc.name}: ${loc.status}
                </li>
            `).join('')}
        </ul>
        <button onclick="backToMain()">Back</button>
    `;
}

// Function to return to main page
function backToMain() {
    location.reload(); // Simple reload for now; can be improved later
}

// Add click events to area cards
document.querySelectorAll('.area-card').forEach(card => {
    card.addEventListener('click', () => {
        const area = card.getAttribute('data-area');
        showArea(area);
    });
});
